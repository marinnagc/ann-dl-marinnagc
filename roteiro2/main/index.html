        import numpy as np
    import matplotlib.pyplot as plt
    from matplotlib import animation

    def set_seed(seed=42):
        np.random.seed(seed)

    def make_dataset(mean0, cov0, mean1, cov1, n_per_class=1000, seed=42):
        set_seed(seed)
        X0 = np.random.multivariate_normal(mean0, cov0, n_per_class)
        X1 = np.random.multivariate_normal(mean1, cov1, n_per_class)
        X = np.vstack([X0, X1])
        y = np.hstack([np.zeros(n_per_class), np.ones(n_per_class)]).astype(int)
        # embaralha
        idx = np.arange(len(X))
        np.random.shuffle(idx)
        return X[idx], y[idx]

    def perceptron_fit(X, y, lr=0.01, max_epochs=100, seed=0):
        """
        Treina perceptron online tradicional (sem classe).
        Retorna um dicion√°rio com hist√≥rico para animar/avaliar.
        Perceptron online: atualiza w,b a cada amostra.
        """
        set_seed(seed) # para reprodutibilidade
        n, d = X.shape # n amostras, d dimens√µes (features)
        w = np.random.randn(d) * 0.01 # inicializa w pequeno
        b = 0.0                     # inicializa b = 0 (bias)

        ws, bs, accs, preds_hist = [], [], [], [] # hist√≥rico por √©poca
        converged = False # se parou antes do max_epochs
        epochs_run = max_epochs # quantas √©pocas rodou de fato 

        for epoch in range(1, max_epochs + 1): 
            updates = 0
            for i in range(n):

                yi = y[i] # label i
                y_hat = 1 if (np.dot(xi, w) + b) >= 0 else 0 # predi√ß√£o (y predito bin√°rio)
                if y_hat != yi: # s√≥ atualiza se errou
                    delta = (yi - y_hat)          # +1 ou -1
                    w = w + lr * delta * xi # atualiza w
                    b = b + lr * delta    # atualiza b
                    updates += 1

            # fim da √©poca: calcula acur√°cia

            scores = X @ w + b 
            '''
            @ em Python faz multiplica√ß√£o de matrizes (produto matricial)
            X: geralmente √© uma matriz (por exemplo, de dados de entrada, onde cada linha √© 
                                        uma amostra e cada coluna √© uma caracter√≠stica/feature).
            w: normalmente √© um vetor de pesos (ou coeficientes) para cada caracter√≠stica.
            b: √© um vi√©s (bias), geralmente um escalar ou vetor.

            Calcula uma combina√ß√£o linear dos dados de entrada (X), ponderada pelos pesos (w),
            e soma o vi√©s (b). O resultado (scores) normalmente representa as predi√ß√µes de um

                                                # usando a fun√ß√£o degrau (step function) do perceptron.

                                    # (y_pred==y) √© um array booleano, compara elemento a elemento,
                                    # True=acertou, False=errou; .mean() d√° a m√©dia de acertos (True=1, False=0)
            # salva hist√≥rico

            # crit√©rio de parada: se n√£o teve atualiza√ß√£o, parou
            # como assim se nao teve atualizacao? se o perceptron acertou todas as amostras
            if updates == 0:
